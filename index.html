I use PDO (Portable Data Object) as my Primary Recipe for PHP Programming / Software Engineering.

$this->_pdo = new PDO("mysql:host=127.0.0.1;dbname=home;","chef","cooker");
$_cook = $this->_pdo -> prepare("SELECT meal AS dinner FROM `home`.`kitchen` WHERE food=:recipe");
$_cook -> execute(["recipe"=>"crabs"]);
$_warez = $_cook -> fetch(PDO::FETCH_ASSOC);
echo($_warez["dinner"]);

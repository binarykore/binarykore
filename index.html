I use PDO (Portable Data Object) as my Primary Recipe for PHP Programming / Software Engineering.


use \PDO;
class x extends \PDO{
 public static function pdo_(){
  return(new PDO("mysql:host=127.0.0.1;dbname=home;","chef","cooker"));
 }
}

$_database = x::pdo_();
$_cook = $_database -> prepare("SELECT meal AS dinner FROM `home`.`kitchen` WHERE food=:recipe");
$_cook -> execute(["recipe"=>"crabs"]);
$_warez = $_cook -> fetch(PDO::FETCH_ASSOC);
echo($_warez["dinner"]);
